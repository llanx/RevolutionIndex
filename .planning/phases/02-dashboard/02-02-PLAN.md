Create `src/components/Gauge.astro` — an Astro component with a `<script>` tag (bundled, NOT `is:inline`) that imports D3 and renders an SVG needle gauge.

**Component structure:**

Frontmatter: empty (no server-side logic needed — data comes via data-* attributes from the parent page).

Template:
```html
<div id="gauge-wrapper">
  <!-- D3 script targets #gauge-mount which is in the parent index.astro -->
</div>
```

Actually, the Gauge component should NOT duplicate the mount div — it should contain the `<script>` tag only, since the mount div with data attributes is already in index.astro. The component's purpose is to encapsulate the D3 rendering script.

Template: Just the `<script>` block. No HTML template needed — the script targets `#gauge-mount` in the parent page.

**Script implementation (bundled `<script>` tag):**

```typescript
import * as d3 from 'd3';

const mount = document.getElementById('gauge-mount');
if (!mount) throw new Error('Gauge mount element not found');

const score = Number(mount.dataset.score);

// Clear placeholder content
mount.innerHTML = '';

// SVG dimensions
const width = 400;
const height = 240;
const cx = width / 2;
const cy = height - 30;  // center near bottom for semicircle
const outerR = 160;
const innerR = 100;

// Zone definitions matching data.ts thresholds and global.css colors
const zones = [
  { start: 0,   end: 25,  color: '#40916c', label: 'Stable' },
  { start: 25,  end: 50,  color: '#f4a261', label: 'Elevated Tension' },
  { start: 50,  end: 75,  color: '#e76f51', label: 'Crisis Territory' },
  { start: 75,  end: 100, color: '#c1121f', label: 'Revolution Territory' },
];

// Map 0-100 score to semicircle angle (radians)
// Left = -PI/2 (score 0), Right = +PI/2 (score 100)
function valueToAngle(v: number): number {
  return ((v / 100) - 0.5) * Math.PI;
}

const arcGen = d3.arc();

const svg = d3.select(mount)
  .append('svg')
  .attr('viewBox', `0 0 ${width} ${height}`)
  .attr('aria-hidden', 'true')  // accessibility label on container div
  .style('width', '100%')
  .style('height', 'auto');

const g = svg.append('g')
  .attr('transform', `translate(${cx},${cy})`);

// Draw zone arcs
zones.forEach(z => {
  g.append('path')
    .attr('d', arcGen({
      innerRadius: innerR,
      outerRadius: outerR,
      startAngle: valueToAngle(z.start),
      endAngle: valueToAngle(z.end),
    }))
    .attr('fill', z.color);
});

// Draw tick marks at zone boundaries (25, 50, 75)
[25, 50, 75].forEach(tick => {
  const angle = valueToAngle(tick);
  const x1 = (outerR + 5) * Math.sin(angle);
  const y1 = -(outerR + 5) * Math.cos(angle);
  const x2 = (outerR - 5) * Math.sin(angle);
  const y2 = -(outerR - 5) * Math.cos(angle);
  g.append('line')
    .attr('x1', x1).attr('y1', y1)
    .attr('x2', x2).attr('y2', y2)
    .attr('stroke', '#666')
    .attr('stroke-width', 2);
});

// Draw needle
const needleAngle = valueToAngle(score);
const needleLength = outerR - 15;
const tipX = needleLength * Math.sin(needleAngle);
const tipY = -needleLength * Math.cos(needleAngle);

// Needle base (triangle shape)
const baseHalfWidth = 6;
const leftAngle = needleAngle - Math.PI / 2;
const rightAngle = needleAngle + Math.PI / 2;
const lx = baseHalfWidth * Math.sin(leftAngle);
const ly = -baseHalfWidth * Math.cos(leftAngle);
const rx = baseHalfWidth * Math.sin(rightAngle);
const ry = -baseHalfWidth * Math.cos(rightAngle);

g.append('path')
  .attr('d', `M ${lx} ${ly} L ${tipX} ${tipY} L ${rx} ${ry} Z`)
  .attr('fill', '#f5f5f5');

// Center pin
g.append('circle')
  .attr('r', 8)
  .attr('fill', '#f5f5f5');

// Min/max labels
g.append('text')
  .attr('x', -(outerR + 15))
  .attr('y', 15)
  .attr('fill', '#a0a0a0')
  .attr('font-size', '12px')
  .attr('text-anchor', 'middle')
  .text('0');

g.append('text')
  .attr('x', outerR + 15)
  .attr('y', 15)
  .attr('fill', '#a0a0a0')
  .attr('font-size', '12px')
  .attr('text-anchor', 'middle')
  .text('100');
```

**Critical implementation notes:**
- Do NOT use `define:vars` — it prevents ES module imports (see RESEARCH.md Pitfall 1)
- Do NOT use `is:inline` — it disables Vite bundling
- The script reads `data-score` from `#gauge-mount` (set by index.astro)
- Zone colors are hardcoded strings matching CSS custom property values (D3 SVG fill attrs don't read CSS custom properties — this is intentional and documented in RESEARCH.md anti-patterns)
- SVG uses `viewBox` so it scales responsively (no fixed pixel dimensions)
- `aria-hidden="true"` on SVG because the container div has `role="img"` and `aria-label`

**Step 1: Create `src/components/TrendChart.astro`**

Like Gauge.astro, this is a `<script>`-only component that targets the `#trend-chart` canvas already in index.astro.

Frontmatter: empty.

Template: Just the `<script>` block (no HTML — canvas is in index.astro).

**Script implementation (bundled `<script>` tag):**

```typescript
import Chart from 'chart.js/auto';

const canvas = document.getElementById('trend-chart') as HTMLCanvasElement;
if (!canvas) throw new Error('Trend chart canvas not found');

const labels: string[] = JSON.parse(canvas.dataset.labels!);
const values: number[] = JSON.parse(canvas.dataset.values!);

// Format date labels for display (e.g., "Dec 7" instead of "2025-12-07")
const displayLabels = labels.map(d => {
  const date = new Date(d + 'T00:00:00');  // Force UTC parse
  return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
});

new Chart(canvas, {
  type: 'line',
  data: {
    labels: displayLabels,
    datasets: [{
      label: 'Revolution Index Score',
      data: values,
      borderColor: '#e85d04',
      backgroundColor: 'rgba(232, 93, 4, 0.1)',
      tension: 0.3,
      fill: true,
      pointRadius: 3,
      pointBackgroundColor: '#e85d04',
    }]
  },
  options: {
    responsive: true,
    maintainAspectRatio: true,
    plugins: {
      legend: { display: false },
      tooltip: {
        mode: 'index',
        intersect: false,
        backgroundColor: '#1a1a1a',
        titleColor: '#f5f5f5',
        bodyColor: '#f5f5f5',
        borderColor: '#333',
        borderWidth: 1,
      }
    },
    scales: {
      x: {
        ticks: {
          maxTicksLimit: 6,
          color: '#a0a0a0',
        },
        grid: {
          color: 'rgba(255, 255, 255, 0.05)',
        }
      },
      y: {
        min: 0,
        max: 100,
        ticks: {
          stepSize: 25,
          color: '#a0a0a0',
        },
        grid: {
          color: 'rgba(255, 255, 255, 0.05)',
        }
      }
    }
  }
});
```
