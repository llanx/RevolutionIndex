export type FactorDirection = 'up' | 'down' | 'neutral';

export interface Factor {
  id: string;
  name: string;
  value: number;       // 0.0-1.0 normalized intensity
  direction: FactorDirection;
  weight: number;      // 0.0-1.0, all weights sum to 1.0

  timestamp: string;   // ISO 8601

  entries: HistoryEntry[];  // { date: string, score: number }[]
}

export interface FactorsData {
  _schema: string;
  factors: FactorDetail[];  // { id, name, description, current_value, historical[] }
}

Frontmatter:
```astro
---
import BaseLayout from '../layouts/BaseLayout.astro';
import FactorsBreakdown from '../components/FactorsBreakdown.astro';
import type { CurrentData, HistoryData, FactorsData } from '../lib/data';

import currentRaw from '../../public/data/current.json';
import historyRaw from '../../public/data/history.json';
import factorsRaw from '../../public/data/factors.json';

const current = currentRaw as unknown as CurrentData;
const history = historyRaw as unknown as HistoryData;
const factors = factorsRaw as unknown as FactorsData;

// Format timestamp for display (static — runs at build time)
const dateDisplay = new Date(current.timestamp).toLocaleDateString('en-US', {
  year: 'numeric',
  month: 'long',
  day: 'numeric',
});

// Enrich current factors with descriptions from factors.json
const enrichedFactors = current.factors.map(f => ({
  ...f,
  description: factors.factors.find(d => d.id === f.id)?.description ?? '',
}));

// Serialize data for client-side charts (Plan 02 will use these)
const chartLabels = JSON.stringify(history.entries.map(e => e.date));
const chartValues = JSON.stringify(history.entries.map(e => e.score));

// Map zone to CSS color variable name for inline styling
const zoneColorMap: Record<string, string> = {
  'Stable': 'var(--zone-stable)',
  'Elevated Tension': 'var(--zone-elevated)',
  'Crisis Territory': 'var(--zone-crisis)',
  'Revolution Territory': 'var(--zone-revolution)',
};
const zoneColor = zoneColorMap[current.zone] ?? 'var(--text-primary)';
---
```
