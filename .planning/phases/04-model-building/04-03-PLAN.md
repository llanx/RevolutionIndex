```python
"""
Ensemble scoring: combine 5 model outputs into a single composite score
using evidence-weighted average.

Weight rationale (from models/README.md):
- PSI 0.25: Strong evidence, longest validation history (Turchin 2003-2023)
- PLI 0.20: Well-validated prospect theory framework for economic stress
- FSP 0.15: Narrower scope (financial crises only) but strongest empirical mechanism (Funke et al. 2016)
- Georgescu SDT 0.25: Most directly applicable to developed democracies (Georgescu 2023)
- V-Dem ERT 0.15: Primarily diagnostic (not predictive), but covers institutional dimension no other model captures
"""
import pandas as pd
import numpy as np
from models import ModelOutput, MODEL_REGISTRY

    """
    Run all registered models and combine their outputs.

    Returns dict with:
    - "composite_score": float (0-100, uncalibrated)
    - "model_outputs": dict[str, ModelOutput]
    - "domain_scores": dict[str, float] (0.0-1.0 per domain)
    - "factor_directions": dict[str, str] (domain_id -> "up"/"down"/"neutral")
    - "history": list[dict] with date and score for each time point
    """
```

Implementation:
1. Import all 5 model modules to trigger @register_model decorators
2. For each model in MODEL_REGISTRY, call the compute function with the unified DataFrame
3. Compute the weighted average: `composite = sum(model.score * MODEL_WEIGHTS[model.model_id] for model in outputs)`
4. Compute domain-level scores from pipeline.compute_domain_scores()
5. Determine factor directions: compare current domain score to 12-month-ago score. If current > previous + 0.02: "up". If current < previous - 0.02: "down". Otherwise: "neutral".
6. Build historical time series: for each month in the unified DataFrame, compute the composite score at that point (running the ensemble on historical slices). For efficiency, compute domain scores for the full history at once, then combine with domain weights.

The historical time series must go back to ~1960 to cover all 6 Phase 5 validation episodes (1968, 1970, 1992, 2001, 2008, 2020). For months where some variables lack data (pre-coverage start), compute the domain score from available variables only (exclude missing, don't treat as zero).

```python
"""
Score calibration: map raw ensemble output to the 0-100 scale using
historical anchor points.

Anchor points (locked decision from CONTEXT.md):
- 2008 financial crisis: should score in Crisis Territory (51-75)
- 2020 (COVID + social unrest): should score in Crisis Territory (51-75)
- Mid-1990s (stability period): should score in Stable (0-25)

Calibration method: linear rescaling based on anchor percentiles.
"""
import numpy as np
import pandas as pd

def calibrate(raw_scores: pd.Series, anchor_scores: dict[str, float] = None) -> pd.Series:
    """
    Calibrate raw ensemble scores to fit anchor-point expectations.

    Default anchors (if not provided):
    - The score at 2008-10 (peak of financial crisis) should be ~65 (mid Crisis Territory)
    - The score at 2020-06 (peak COVID + BLM) should be ~65 (mid Crisis Territory)
    - The average score for 1994-1997 should be ~20 (mid Stable)

    Method:
    1. Compute the raw scores at anchor dates
    2. Fit a linear transformation: calibrated = a * raw + b
       such that anchor dates map to target values
    3. Clamp to 0-100

    Returns calibrated pd.Series.
    """

def compute_bootstrap_ci(unified_df: pd.DataFrame, n_bootstrap: int = 1000,
                         ci_width: float = 0.90) -> dict:
    """
    Bootstrap confidence intervals for the composite score.

    Method:
    1. Resample variables (columns) with replacement n_bootstrap times
    2. For each resample, recompute domain scores and ensemble composite
    3. Extract ci_width percentile interval

    Returns dict with:
    - "point_estimate": float
    - "ci_lower": float (5th percentile for 90% CI)
    - "ci_upper": float (95th percentile for 90% CI)
    - "n_bootstrap": int
    - "ci_width": float

    Required by Phase 5 TEST-03.
    """

def score_to_zone(score: int) -> str:
    """
    Map a 0-100 integer score to zone label.

    0-25: "Stable"
    26-50: "Elevated Tension"
    51-75: "Crisis Territory"
    76-100: "Revolution Territory"

    Uses the ZONES constant boundaries from data.ts.
    """
```

Implement all three functions.

Calibration approach:
- Compute raw scores for the full history
- Identify anchor points: 2008-10 raw score, 2020-06 raw score, 1994-1997 average raw score
- Fit linear transformation: solve for a, b such that `a * raw_crisis + b = 65` and `a * raw_stable + b = 20`
- Apply to full history
- Clamp to 0-100

Bootstrap CI approach:
- Resample the 41 variables (columns) with replacement
- For each resample, recompute evidence-weighted domain scores and ensemble composite
- This captures uncertainty from both variable selection and evidence weighting
- Use 1000 iterations (sufficient for 90% CI stability)
- Store bootstrap distribution for Phase 5 use

**Anti-patterns to avoid:**
- Do NOT hardcode calibration parameters that can't be traced to anchor points
- Do NOT use parametric confidence intervals (bootstrap is non-parametric, appropriate for non-normal distributions)
- Do NOT compute CIs by perturbing the composite score directly -- resample at the variable level to propagate uncertainty correctly
- Do NOT produce flat history (data.ts requires score variation of at least 10 points across the series)

**2. models/run.py -- End-to-End Pipeline Entry Point:**

```python
"""
End-to-end pipeline: fetch data -> normalize -> run models -> ensemble -> calibrate -> write JSON.

Usage:
    python models/run.py                    # full pipeline (fetches data)
    python models/run.py --cached-only      # skip API fetch, use cached data only
    python models/run.py --dry-run          # compute scores but don't write JSON

Requires FRED_API_KEY environment variable (free key from https://fred.stlouisfed.org/docs/api/api_key.html)
"""
import argparse
import os
import sys
from datetime import datetime, timezone

def main():
    parser = argparse.ArgumentParser(description="Revolution Index Pipeline")
    parser.add_argument("--cached-only", action="store_true",
                        help="Skip API fetching, use cached data only")
    parser.add_argument("--dry-run", action="store_true",
                        help="Compute scores but don't write JSON output")
    args = parser.parse_args()

    # 1. Get FRED API key
    api_key = os.environ.get("FRED_API_KEY")
    if not api_key and not args.cached_only:
        print("ERROR: FRED_API_KEY environment variable not set.")
        print("Get a free key at: https://fred.stlouisfed.org/docs/api/api_key.html")
        sys.exit(1)

    # 2. Fetch and prepare data
    from pipeline import fetch_all, compute_domain_scores
    unified_df = fetch_all(api_key=api_key or "", start_year=1947)
    domain_scores_df = compute_domain_scores(unified_df)

    # 3. Run all models
    from ensemble import compute_ensemble
    ensemble_result = compute_ensemble(unified_df)

    # 4. Calibrate scores
    from calibrate import calibrate, compute_bootstrap_ci, score_to_zone
    import pandas as pd
    raw_history = pd.Series(
        {row["date"]: row["score"] for row in ensemble_result["history"]})
    calibrated_history = calibrate(raw_history)

    # 5. Compute bootstrap CI for current score
    bootstrap_ci = compute_bootstrap_ci(unified_df)

    # 6. Determine current values
    current_score = int(round(calibrated_history.iloc[-1]))
    current_zone = score_to_zone(current_score)
    timestamp = datetime.now(timezone.utc).isoformat()

    # 7. Print summary
    print(f"Composite Score: {current_score}")
    print(f"Zone: {current_zone}")
    print(f"Bootstrap 90% CI: [{bootstrap_ci['ci_lower']:.1f}, {bootstrap_ci['ci_upper']:.1f}]")
    print(f"Models: {', '.join(ensemble_result['model_outputs'].keys())}")
    for domain_id, score in ensemble_result["domain_scores"].items():
        print(f"  {domain_id}: {score:.3f}")

    # 8. Write JSON output
    if not args.dry_run:
        from output import write_current_json, write_history_json, write_factors_json
        write_current_json(
            composite_score=current_score,
            zone=current_zone,
            domain_scores=ensemble_result["domain_scores"],
            domain_directions=ensemble_result["factor_directions"],
            domain_weights={d.value: w for d, w in
                          __import__("config").DOMAIN_WEIGHTS.items()},
            timestamp=timestamp,
            bootstrap_ci=bootstrap_ci,
        )
        # Build history for JSON (sample annually pre-2000, quarterly post-2000)
        history_entries = build_history_entries(calibrated_history)
        write_history_json(history_entries)
        # Build factor history for sparklines
        write_factors_json(
            domain_scores=ensemble_result["domain_scores"],
            domain_history=build_domain_history(domain_scores_df),
        )
        print(f"\nJSON files written to public/data/")

    from pipeline import get_freshness
    freshness = get_freshness()
    print(f"\nData freshness: {len(freshness)} sources tracked")
```
